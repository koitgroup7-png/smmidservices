function doPost(e) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const data = JSON.parse(e.postData.contents);

  let usersSheet = ss.getSheetByName("Users");
  let ordersSheet = ss.getSheetByName("Orders");
  let stockSheet = ss.getSheetByName("Stock");

  if (!usersSheet) {
    usersSheet = ss.insertSheet("Users");
    usersSheet.appendRow(["User ID", "Username", "Name", "First Visit", "Last Visit", "Total Visits", "Total Orders"]);
  }

  if (!ordersSheet) {
    ordersSheet = ss.insertSheet("Orders");
    ordersSheet.appendRow(["Order ID", "User ID", "Username", "Name", "Product", "Quantity", "Area Code", "Price", "Payment Status", "Delivery Status", "Date"]);
  }

  if (!stockSheet) {
    stockSheet = ss.insertSheet("Stock");
    stockSheet.appendRow(["Stock ID", "Product Name", "Area Code", "Stock Data", "Status", "Sold To", "Username", "Order ID", "Sold At"]);
  }

  if (data.type === "user_visit" || data.type === "terms") {
    updateUser(usersSheet, data);
  }

  if (data.type === "order") {
    addOrder(usersSheet, ordersSheet, data);
  }

  if (data.type === "take_stock_bulk") {
    return takeStockBulk(stockSheet, data);
  }

  return jsonOutput({ success: true });
}

function updateUser(sheet, data) {
  const userId = String(data.user_id || "");
  const values = sheet.getDataRange().getValues();
  const now = new Date();

  for (let i = 1; i < values.length; i++) {
    if (String(values[i][0]) === userId) {
      sheet.getRange(i + 1, 2).setValue(data.username || "");
      sheet.getRange(i + 1, 3).setValue(data.name || "");
      sheet.getRange(i + 1, 5).setValue(now);
      sheet.getRange(i + 1, 6).setValue(Number(values[i][5] || 0) + 1);
      return;
    }
  }

  sheet.appendRow([userId, data.username || "", data.name || "", now, now, 1, 0]);
}

function addOrder(usersSheet, ordersSheet, data) {
  const now = new Date();

  ordersSheet.appendRow([
    data.order_id || "",
    data.user_id || "",
    data.username || "",
    data.name || "",
    data.product || "",
    data.quantity || "",
    data.area_code || "",
    data.price || "",
    data.payment_status || "Paid",
    data.delivery_status || "Delivered",
    now
  ]);

  const userId = String(data.user_id || "");
  const values = usersSheet.getDataRange().getValues();

  for (let i = 1; i < values.length; i++) {
    if (String(values[i][0]) === userId) {
      usersSheet.getRange(i + 1, 2).setValue(data.username || "");
      usersSheet.getRange(i + 1, 3).setValue(data.name || "");
      usersSheet.getRange(i + 1, 5).setValue(now);
      usersSheet.getRange(i + 1, 7).setValue(Number(values[i][6] || 0) + 1);
      return;
    }
  }

  usersSheet.appendRow([userId, data.username || "", data.name || "", now, now, 1, 1]);
}

function takeStockBulk(stockSheet, data) {
  const lock = LockService.getScriptLock();
  lock.waitLock(30000);

  try {
    const productName = String(data.product_name || "").trim().toLowerCase();
    const areaCode = String(data.area_code || "Random").trim().toLowerCase();
    const quantity = Number(data.quantity || 1);
    const orderId = String(data.order_id || "");
    const userId = String(data.user_id || "");
    const username = String(data.username || "");
    const now = new Date();

    const values = stockSheet.getDataRange().getValues();
    const selected = [];

    for (let i = 1; i < values.length; i++) {
      const rowProduct = String(values[i][1] || "").trim().toLowerCase();
      const rowArea = String(values[i][2] || "random").trim().toLowerCase();
      const rowStockData = String(values[i][3] || "");
      const rowStatus = String(values[i][4] || "").trim().toLowerCase();

      const productMatch = rowProduct === productName;
      const areaMatch = !areaCode || areaCode === "random" || rowArea === areaCode || rowArea === "random";
      const available = rowStatus === "available";

      if (productMatch && areaMatch && available) {
        selected.push({
          rowNumber: i + 1,
          stock_id: values[i][0],
          stock_data: rowStockData
        });

        if (selected.length >= quantity) break;
      }
    }

    if (selected.length < quantity) {
      return jsonOutput({
        success: false,
        message: "No available stock found"
      });
    }

    for (const item of selected) {
      stockSheet.getRange(item.rowNumber, 5).setValue("Sold");
      stockSheet.getRange(item.rowNumber, 6).setValue(userId);
      stockSheet.getRange(item.rowNumber, 7).setValue(username);
      stockSheet.getRange(item.rowNumber, 8).setValue(orderId);
      stockSheet.getRange(item.rowNumber, 9).setValue(now);
    }

    return jsonOutput({
      success: true,
      stocks: selected.map(item => ({
        stock_id: item.stock_id,
        stock_data: item.stock_data
      }))
    });
  } finally {
    lock.releaseLock();
  }
}

function jsonOutput(obj) {
  return ContentService
    .createTextOutput(JSON.stringify(obj))
    .setMimeType(ContentService.MimeType.JSON);
}
